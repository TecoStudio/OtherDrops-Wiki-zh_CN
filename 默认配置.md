#######################
# 配置文件 #############
# Translated ##########
#######################
# 将其设置为"low"以减少引导消息的数量或设置为 high, highest, extreme 以获得更多调试信息
verbosity: normal

#######################
# 将钱四舍五入到的小数位数（对于掉落和消息）
money-precision: 2

#######################
# 禁用匿名日志记录 - 如果您在启动时在旧版 Tekkit 或 Bukkit 中遇到错误，请将此设置为 true
disable_metrics: false

#######################
# 启用 WorldGuard BUILD 权限？ - 如果你想让掉落物只在玩家可以建造的地方运行，请将其设置为 true
enable_wg_matching: false

#######################
# 我们应该检查更新吗？ - 如果您不想收到新版本的通知，请将此设置为 false
update_checker: true

#######################
# 物品应该掉到地上吗？ - 如果您希望物品直接进入玩家的库存，请将其设置为 false
item_falls_on_ground: true

#######################
# 警告：只有在您知道自己在做什么的情况下才能更改此设置！！！
# 将其更改为 true 将允许从方块和物品交互中删除准备好的TNT。
# 通常，该插件只允许从怪物/动物身上掉落准备好的 TNT。 改变这个
# 为 true 可能会导致一连串不间断的 TNT，并可能最终导致服务器崩溃！
# 只有有经验的用户才应该将其更改为 true!!!
primed_tnt: false

#######################
# 如果标准 config/include 文件被删除，则恢复它们
restore_deleted_config_files: true



#######################
# 附魔          ########
#######################
# Ignore level: 允许附魔忽略开始和/或最高等级（例如 dig_speed#50）
# 注意：enchantments_ignore_level 要求 enchantments_use_unsafe 为 true

enchantments_ignore_level: false

# 不安全使用：允许对任何物品/工具附魔（例如 DIRT@!DAMAGE_ALL#3）

enchantments_use_unsafe: false

# 限制匹配：如果为真，则要求附魔完全符合它们在掉落文件中的配置方式
#（例如，用 Fortune 3 定义的镐只会让 Fortune 3 的镐执行掉落。要允许时运3、耐久3和效率5的镐执行掉落，# 则请将此值设置为 false。

enchantments_restrict_matching: true
####



#######################
# 掉落  ###############
#######################
# 如果您不希望方块在爆炸时丢弃它们的自定义配置（例如 tnt/creepers），请将其设置为 false
customdropsforexplosions: true

#######################
# 如果您希望物品作为单个堆叠掉落，请将此设置为 false（true=单个物品掉落，如青金石矿石中的青金石掉落）
default_dropspread: true

#######################
# 如果您希望在配置生物掉落时取消 XP 掉落，请将此设置为 true（除非有默认掉落）
# - 这意味着如果你想要任何 XP 掉落，你需要自己设置一个 XP 掉落
disable_xp_on_non_default: false

#######################
# 允许任何替换方块（默认情况下，不安全的替换，例如项目，会被阻止，因为它们会使客户端崩溃）
allow_any_replacementblock: false

#######################
# 不放弃默认值的自定义块中断事件被取消，因此 mcMMO 不会为该中断生成 XP
# 此设置告诉 OtherDrops 手动将这些取消的事件发送到 mcMMO，收集 mcMMO-XP 和任何双倍掉落等。
send_customblockbreak_to_mcmmo: true

#######################
##     覆盖默认值     ##
##
money_overrides_default: false  # 设置为 true 让钱掉落覆盖默认掉落
xp_overrides_default: false     # 设置 true 让 xp 掉落覆盖默认掉落
loot_overrides_default: true    # 设置为 true 以拥有战利品（除金钱或 xp 之外的任何东西，包括生物）掉落覆盖默认掉落
####



#######################
# 杂项 ################
#######################
# 覆盖爆炸上限（通常上限为最大功率 100f 以避免服务器崩溃）
override_explosion_cap: false
# Action radius - 考虑消息、药水效果、声音、伤害和 pset 动作的半径
action_radius: 10

#######################
# 彩色日志消息 - 如果它们惹恼了您，请将其禁用
color_log_messages: true

#######################
# 如果当前世界中的生物数量较多，OtherDrops 将停止生成新的生物
# 这只会影响 mobspawn 触发器
custom_spawn_limit: 300



#######################
#    时间和日期格式   ##
#######################
# 对于消息变量 (${time} & ${date})
# 请参见 http://docs.oracle.com/javase/1.4.2/docs/api/java/text/SimpleDateFormat.html
# 获取详细用法
#
time_format: "HH:mm:ss"
date_format: "yyyy/MM/dd"
